使用说明：


<div id="J_pagination"><div/>


//初始化
$('#J_pagination').tfPagination({
    totalCount: 80,
    currentPage: 4,
    pageSize: 10,
    maxLabelNumber: 7, //最多展示的页码数
    handler: function(currentPage,pageSize){  //currentPage发生变化时触发的函数
         alert('currentPage:'+currentPage+'/n'+'paseSize:'+pageSize);
    }
});
